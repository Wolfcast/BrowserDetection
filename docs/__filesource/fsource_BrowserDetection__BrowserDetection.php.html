  199.     /**
  200.      * Compare two version number strings.
  201.      * @param string $sourceVer The source version number.
  202.      * @param string $compareVer The version number to compare with the source version number.
  203.      * @return int Returns 1 if $sourceVer < $compareVer, 0 if $sourceVer == $compareVer or -1 if $sourceVer > $compareVer.
  204.      */
  205.     public function compareVersions($sourceVer, $compareVer)
  206.     {
  207.         $sourceVer = explode('.', $sourceVer);
  208.         foreach ($sourceVer as $k => $v) {
  209.             $sourceVer[$k] = $this->parseInt($v);
  210.         }
  211.  
  212.         $compareVer = explode('.', $compareVer);
  213.         foreach ($compareVer as $k => $v) {
  214.             $compareVer[$k] = $this->parseInt($v);
  215.         }
  216.  
  217.         if (count($sourceVer) != count($compareVer)) {
  218.             if (count($sourceVer) > count($compareVer)) {
  219.                 for ($i = count($compareVer); $i < count($sourceVer); $i++) {
  220.                     $compareVer[$i] = 0;
  221.                 }
  222.             } else {
  223.                 for ($i = count($sourceVer); $i < count($compareVer); $i++) {
  224.                     $sourceVer[$i] = 0;
  225.                 }
  226.             }
  227.         }
  228.  
  229.         foreach ($sourceVer as $i => $srcVerPart) {
  230.             if ($srcVerPart > $compareVer[$i]) {
  231.                 return -1;
  232.             } else {
  233.                 if ($srcVerPart < $compareVer[$i]) {
  234.                     return 1;
  235.                 }
  236.             }
  237.         }
  238.  
  239.         return 0;
  240.     }
