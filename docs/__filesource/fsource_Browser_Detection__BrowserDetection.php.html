  449.     /**
  450.      * Compare two version number strings.
  451.      * @param string $sourceVer The source version number.
  452.      * @param string $compareVer The version number to compare with the source version number.
  453.      * @return int Returns -1 if $sourceVer < $compareVer, 0 if $sourceVer == $compareVer or 1 if $sourceVer >
  454.      *  $compareVer.
  455.      */
  456.     public function compareVersions($sourceVer, $compareVer)
  457.     {
  458.         $sourceVer = explode('.', $sourceVer);
  459.         foreach ($sourceVer as $k => $v) {
  460.             $sourceVer[$k] = $this->parseInt($v);
  461.         }
  462.  
  463.         $compareVer = explode('.', $compareVer);
  464.         foreach ($compareVer as $k => $v) {
  465.             $compareVer[$k] = $this->parseInt($v);
  466.         }
  467.  
  468.         if (count($sourceVer) != count($compareVer)) {
  469.             if (count($sourceVer) > count($compareVer)) {
  470.                 for ($i = count($compareVer); $i < count($sourceVer); $i++) {
  471.                     $compareVer[$i] = 0;
  472.                 }
  473.             } else {
  474.                 for ($i = count($sourceVer); $i < count($compareVer); $i++) {
  475.                     $sourceVer[$i] = 0;
  476.                 }
  477.             }
  478.         }
  479.  
  480.         foreach ($sourceVer as $i => $srcVerPart) {
  481.             if ($srcVerPart > $compareVer[$i]) {
  482.                 return 1;
  483.             } else {
  484.                 if ($srcVerPart < $compareVer[$i]) {
  485.                     return -1;
  486.                 }
  487.             }
  488.         }
  489.  
  490.         return 0;
  491.     }
